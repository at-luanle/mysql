### 24. Lấy 1 blog và 1 news có số lượng comment nhiều nhất
```mysql
(	select  id, title
	from blog , (select target_id, max(mycount)
						from ( select target_id,target_table,count(target_id)  as mycount
									from comment group by target_id
									having target_table = 'blog'  )  as counts ) as c
						where id  = c.target_id
)
union
(	select  id, title
	from news , (select target_id, max(mycount)
						from ( select target_id,target_table,count(target_id)  as mycount
									from comment group by target_id
									having target_table = 'news'  )  as counts ) as c
						where id  = c.target_id
)

  ````
