### 24. Lấy 1 blog và 1 news có số lượng comment nhiều nhất
```mysql
  select blog.*, news.*
  from news
  join category on news.category_id = category.id
  join blog on blog.category_id = category.id
  join user on blog.user_id = (
    select user.id
    from user
    join (select user_id
      from comment
      group by user_id
      having count(user_id) = (
        select max(mycount)
        from (
          select user_id, count(*) as mycount
          from comment
          group by user_id) as counts))
  as u on u.user_id = user.id
  )
  limit 1;
  ````
